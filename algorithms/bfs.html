🌊 BFS (Breadth-First Search)

Обход графа в ширину - поиск кратчайшего пути в невзвешенном графе

🐍 Python
from collections import deque

def bfs(graph, start):
    visited = set()
    queue = deque([start])
    visited.add(start)
    result = []
    
    while queue:
        vertex = queue.popleft()
        result.append(vertex)
        
        for neighbor in graph[vertex]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)
    
    return result

# Пример использования
graph = {
    'A': ['B', 'C'],
    'B': ['D', 'E'],
    'C': ['F'],
    'D': [],
    'E': ['F'],
    'F': []
}
print(bfs(graph, 'A'))  # ['A', 'B', 'C', 'D', 'E', 'F']
⚙️ C++
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <unordered_map>

using namespace std;

vector<string> bfs(unordered_map<string, vector<string>>& graph, string start) {
    unordered_set<string> visited;
    queue<string> q;
    vector<string> result;
    
    visited.insert(start);
    q.push(start);
    
    while (!q.empty()) {
        string vertex = q.front();
        q.pop();
        result.push_back(vertex);
        
        for (string neighbor : graph[vertex]) {
            if (visited.find(neighbor) == visited.end()) {
                visited.insert(neighbor);
                q.push(neighbor);
            }
        }
    }
    
    return result;
}

int main() {
    unordered_map<string, vector<string>> graph = {
        {"A", {"B", "C"}},
        {"B", {"D", "E"}},
        {"C", {"F"}},
        {"D", {}},
        {"E", {"F"}},
        {"F", {}}
    };
    
    vector<string> result = bfs(graph, "A");
    for (string v : result) {
        cout << v << " ";
    }
    return 0;
}

📖 Как работает BFS?

BFS (Breadth-First Search) - алгоритм обхода графа, который исследует все вершины на текущем уровне, прежде чем перейти на следующий.

  • 🔄 Использует очередь (Queue)
  • 📍 Находит кратчайший путь в невзвешенном графе
  • ⏱️ Временная сложность: O(V + E), где V - вершины, E - рёбра
  • 💾 Пространственная сложность: O(V)

Применение: Поиск кратчайшего пути в лабиринте, социальные сети (поиск друзей), веб-краулеры.
